The objective of conjoint analysis is to measure how respondents trade-off various alternatives and their respective attribute levels; for example, the trade-off a consumer faces is whether to buy a cheaper car with higher fuel consumption or a more expensive one with lower fuel consumption. Conjoint analysis is a survey technique that asks respondents to evaluate products. These product evaluations are then decomposed to derive estimates on the utilities of the products constituent attribute levels (the part-worths). Conjoint analysis is thus a decompositional approach.

Researchers can estimate part-worths based on estimation techniques such as regression analysis (ratings-based data) or multinomial logistic regression (choice data). The total value of an alternative can then be computed as the sum of its respective part-worth utilities. Almost all published research on conjoint analysis and preference measurement is based on the assumption of a linear additive utility model.

# Stan model recieves number of options in each variable (as list)
n_vars <- cbc_saw %>% 
    summarise_if(is.numeric, max) %>% 
    select(-choice) %>% 
    map(~.)
names_vars_n <- str_c("n_", str_remove(names(n_vars), "_int"))

# index matrix for iterations
index <- array(0, dim = c(n_vars$n_resp_id, n_vars$n_ques, n_vars$n_alt))

r <- 1
for(i in 1:n_vars$n_resp_id){
    for(j in 1:n_vars$n_ques){
        for(k in 1:n_vars$n_alt){
            index[i, j, k] <- r
            r <- r + 1
        }
    }
}

vars <- map(cbc_saw, ~.)
params_stan <- c(N = nrow(cbc_saw), n_vars, vars, list(index = index))

get_summary_coef <- function(fit, data_levels = cbc_levels){
    summary_fit <- summary(fit)$summary %>% 
        as_tibble(rownames = "coef") %>% 
        mutate(
            var = str_extract(coef, pattern = "_([a-z]+)_|_([a-z]+)") %>%
                str_remove_all("_"),
            level = str_extract(coef, pattern = "[0-9]+(])$") %>% 
                str_remove_all("]") %>% as.integer(), 
            ind = str_extract(coef, pattern = "[0-9]+(,)") %>% 
                str_remove_all(",") %>% as.integer()
            ) %>%
        filter(!is.na(var), !is.na(level)) %>% 
        left_join(cbc_levels, by = c("var", "level")) 
}

summary_fit_plot <- get_summary_coef(fit_1, cbc_levels)
